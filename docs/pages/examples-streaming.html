Example

Live Streaming

Player movement replication at 20 Hz using Stream's binary delta-compressed channels. The server re-broadcasts positions via Broadcast, and clients interpolate smoothly between received frames.

How it works

1
Client sends CFrame + velocity at 20 Hz over UnreliableRemoteEvent via channel:Send()
2
Stream delta-compresses the packet | only changed fields are sent in full
3
Server receives via channel:Subscribe(), stores position, re-broadcasts to other players
4
Other clients receive via listener:On("pos"), push to interpolation buffer
5
Client interpolates between buffered frames on RunService.Heartbeat

Shared schema

-- ReplicatedStorage/Schemas/PlayerState.luau
local RoExpress = require(game.ReplicatedStorage.RoExpress)
return RoExpress.Stream:Schema({
    cf       = "CFrame",   -- root part CFrame
    vel      = "Vector3", -- linear velocity
    anim     = "uint8",   -- animation state ID
    health   = "uint8",   -- 0-100
})

Server: MovementServer.luau

local RoExpress = require(game.ReplicatedStorage.RoExpress)
local Schema    = require(game.ReplicatedStorage.Schemas.PlayerState)
local broadcast = RoExpress("Broadcast")
local channel   = RoExpress.Stream:Channel("playerState", Schema)

channel:Subscribe(function(player, data)
    -- Fan out to everyone except the sender
    local others = {}
    for _, p in game.Players:GetPlayers() do
        if p ~= player then
            others[#others + 1] = p
        end
    end
    broadcast:EmitTo(others, "pos", {
        id   = player.UserId,
        cf   = data.cf,
        vel  = data.vel,
        anim = data.anim,
    })
end)

Client: MovementClient.luau

local RunService = game:GetService("RunService")
local Schema     = require(game.ReplicatedStorage.Schemas.PlayerState)
local channel    = RoExpress.Stream:Channel("playerState", Schema)
local listener   = RoExpress("Listener")
local buffers    = {}  -- { [userId] = { cf, vel, t } }
local localChar  = game.Players.LocalPlayer.Character

-- Send own state at 20 Hz
local accum = 0
RunService.Heartbeat:Connect(function(dt)
    accum += dt
    if accum < 0.05 then return end
    accum = 0
    if not localChar then return end
    local hrp = localChar:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    channel:Send({
        cf     = hrp.CFrame,
        vel    = hrp.AssemblyLinearVelocity,
        anim   = getCurrentAnim(),
        health = math.floor(game.Players.LocalPlayer.Character.Humanoid.Health),
    })
end)

-- Receive others' positions
listener:On("pos", function(data)
    buffers[data.id] = { cf = data.cf, vel = data.vel, t = tick() }
end)

-- Interpolate on every frame
RunService.Heartbeat:Connect(function(dt)
    for userId, buf in buffers do
        local char = getCharForUserId(userId)
        if char then
            smoothMove(char, buf.cf, buf.vel, dt)
        end
    end
end)
