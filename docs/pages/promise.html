Async / Promise client

Network callbacks are the default. Wrapping them in Promises gives you chaining, parallel-merge, and cancellation for complex async flows.

No built-in Promise library. RoExpress doesn't ship a Promise implementation — use evaera's roblox-lua-promise (Wally: evaera/promise@4) or any library with the same API.

Wrapping a Network call

local Promise = require(ReplicatedStorage.Packages.Promise)

local function pGet(route, data)
    return Promise.new(function(resolve, reject)
        network:Get(route, data, function(res)
            if res.ok then resolve(res.data)
            else           reject(res.status) end
        end)
    end)
end

local function pPost(route, data)
    return Promise.new(function(resolve, reject)
        network:Post(route, data, function(res)
            if res.ok then resolve(res.data)
            else           reject(res.status) end
        end)
    end)
end

Chain

pGet("shop/catalogue")
    :andThen(function(data)
        ShopUI:Load(data)
        return pGet("player/data")    -- second request after first resolves
    end)
    :andThen(function(data)
        CoinsUI:Set(data.coins)
    end)
    :catch(function(status)
        UI:ShowError("Load failed: " .. status)
    end)

Parallel

Promise.all({
    pGet("shop/catalogue"),
    pGet("player/data"),
    pGet("leaderboard"),
})
    :andThen(function(results)
        UI:LoadAll(results[1], results[2], results[3])
    end)
    :catch(function(err) UI:ShowError(err) end)

Cancel on cleanup

local loading = Promise.all({ pGet("shop/catalogue"), pGet("player/data") })
    :andThen(function(results) UI:LoadAll(results) end)

game.Players.LocalPlayer.AncestryChanged:Connect(function()
    loading:cancel()   -- clean up if player leaves
end)

Wrapping Listener:Once

local function onceEvent(event)
    return Promise.new(function(resolve)
        listener:Once(event, resolve)
    end)
end

onceEvent("player.data"):andThen(function(data)
    UI:SetCoins(data.coins)
end)
